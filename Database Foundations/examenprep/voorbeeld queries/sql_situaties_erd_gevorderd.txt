
DATABASE OEFENING – 30 SITUATIES (GEAVANCEERD)
============================================
GEDEELTE 1 – 20 COMPLEXERE QUERY-SITUATIES
-----------------------------------------

1. Situatie:
Geef per klant het aantal dieren dat hij/zij bezit, gesorteerd van veel naar weinig.
SQL:
SELECT customer.first_name, customer.last_name, COUNT(animal.id) AS aantal_dieren
FROM customer
JOIN animal ON customer.id = animal.customer_id
GROUP BY customer.first_name, customer.last_name
ORDER BY aantal_dieren DESC;

2. Situatie:
Geef alle dieren die nog nooit een afspraak hebben gehad.
SQL:
SELECT animal.name
FROM animal
LEFT JOIN appointment ON animal.id = appointment.animal_id
WHERE appointment.id IS NULL;

3. Situatie:
Geef per medewerker het aantal afspraken waaraan hij/zij gekoppeld is.
SQL:
SELECT employee.first_name, employee.last_name, COUNT(employee_appointment.appointment_id) AS aantal_afspraken
FROM employee
LEFT JOIN employee_appointment ON employee.id = employee_appointment.employee_id
GROUP BY employee.first_name, employee.last_name;

4. Situatie:
Toon alle afspraken waarbij meer dan één medewerker betrokken is.
SQL:
SELECT appointment.id, COUNT(employee_appointment.employee_id) AS aantal_medewerkers
FROM appointment
JOIN employee_appointment ON appointment.id = employee_appointment.appointment_id
GROUP BY appointment.id
HAVING COUNT(employee_appointment.employee_id) > 1;

5. Situatie:
Geef per diersoort het aantal afspraken.
SQL:
SELECT species.species, COUNT(appointment.id) AS aantal_afspraken
FROM appointment
JOIN animal ON appointment.animal_id = animal.id
JOIN species ON animal.species_id = species.id
GROUP BY species.species;

6. Situatie:
Geef alle klanten die minstens één afspraak aan huis hebben gehad.
SQL:
SELECT DISTINCT customer.first_name, customer.last_name
FROM customer
JOIN animal ON customer.id = animal.customer_id
JOIN appointment ON animal.id = appointment.animal_id
WHERE appointment.address_home_visit IS NOT NULL;

7. Situatie:
Geef het totaalbedrag dat per consult werd aangerekend (consult + aankopen).
SQL:
SELECT consult.id,
       consult.price + COALESCE(SUM(purchase.price * purchase.amount), 0) AS totaalbedrag
FROM consult
LEFT JOIN purchase ON consult.id = purchase.consult_id
GROUP BY consult.id, consult.price;

8. Situatie:
Geef alle producten die nog nooit verkocht zijn.
SQL:
SELECT product.name
FROM product
LEFT JOIN purchase ON product.id = purchase.product_id
LEFT JOIN purchase_at_reception ON product.id = purchase_at_reception.product_id
WHERE purchase.product_id IS NULL
AND purchase_at_reception.product_id IS NULL;

9. Situatie:
Geef per producttype de totale omzet.
SQL:
SELECT product_type.type_name, SUM(purchase.amount * purchase.price) AS omzet
FROM purchase
JOIN product ON purchase.product_id = product.id
JOIN product_type ON product.product_type = product_type.type_name
GROUP BY product_type.type_name;

10. Situatie:
Geef medewerkers die zowel technische expertise hebben als met minstens één diersoort werken.
SQL:
SELECT DISTINCT employee.first_name, employee.last_name
FROM employee
JOIN vet_technical_expertise ON employee.id = vet_technical_expertise.employee_id
JOIN species_employee ON employee.id = species_employee.employee_id;

11. Situatie:
Geef de oudste dieren per diersoort.
SQL:
SELECT species.species, MIN(animal.year_of_birth) AS oudste
FROM animal
JOIN species ON animal.species_id = species.id
GROUP BY species.species;

12. Situatie:
Geef alle afspraken die gepland zijn in de toekomst.
SQL:
SELECT appointment.id, appointment.date_time
FROM appointment
WHERE appointment.date_time > CURRENT_TIMESTAMP;

13. Situatie:
Geef per klant het totaal uitgegeven bedrag (consulten + balieverkoop).
SQL:
SELECT customer.first_name, customer.last_name,
       COALESCE(SUM(consult.price),0) + COALESCE(SUM(purchase_at_reception.price * purchase_at_reception.amount),0) AS totaal
FROM customer
LEFT JOIN animal ON customer.id = animal.customer_id
LEFT JOIN appointment ON animal.id = appointment.animal_id
LEFT JOIN consult ON appointment.id = consult.id
LEFT JOIN purchase_at_reception ON customer.id = purchase_at_reception.customer_id
GROUP BY customer.first_name, customer.last_name;

14. Situatie:
Geef medewerkers die nog geen beschikbaarheden hebben.
SQL:
SELECT employee.first_name, employee.last_name
FROM employee
LEFT JOIN employee_availability ON employee.id = employee_availability.employee_id
WHERE employee_availability.employee_id IS NULL;

15. Situatie:
Geef alle dieren waarvan de eigenaar geen telefoonnummer heeft.
SQL:
SELECT animal.name
FROM animal
JOIN customer ON animal.customer_id = customer.id
WHERE customer.phone IS NULL;

16. Situatie:
Geef alle producten waarvan de voorraad lager is dan het gemiddelde.
SQL:
SELECT product.name
FROM product
WHERE product.stock_amount < (
    SELECT AVG(product.stock_amount) FROM product
);

17. Situatie:
Geef per medewerker het aantal technische expertises.
SQL:
SELECT employee.first_name, employee.last_name, COUNT(vet_technical_expertise.technical_expertise_id) AS aantal
FROM employee
LEFT JOIN vet_technical_expertise ON employee.id = vet_technical_expertise.employee_id
GROUP BY employee.first_name, employee.last_name;

18. Situatie:
Geef afspraken zonder gekoppelde medewerkers.
SQL:
SELECT appointment.id
FROM appointment
LEFT JOIN employee_appointment ON appointment.id = employee_appointment.appointment_id
WHERE employee_appointment.employee_id IS NULL;

19. Situatie:
Geef klanten die meer dan één dier hebben.
SQL:
SELECT customer.first_name, customer.last_name
FROM customer
JOIN animal ON customer.id = animal.customer_id
GROUP BY customer.first_name, customer.last_name
HAVING COUNT(animal.id) > 1;

20. Situatie:
Geef producten die zowel bij consulten als aan de balie verkocht worden.
SQL:
SELECT DISTINCT product.name
FROM product
JOIN purchase ON product.id = purchase.product_id
JOIN purchase_at_reception ON product.id = purchase_at_reception.product_id;


GEDEELTE 2 – 10 AANPASSINGEN / TOEVOEGINGEN (DML)
------------------------------------------------

21. Situatie:
Voeg een nieuwe klant toe.
SQL:
INSERT INTO customer (first_name, last_name, town, email)
VALUES ('Jan', 'Peeters', 'Gent', 'jan.peeters@mail.be');

22. Situatie:
Voeg een nieuw dier toe voor een bestaande klant.
SQL:
INSERT INTO animal (species_id, customer_id, name, sex)
VALUES (1, 3, 'Max', 'M');

23. Situatie:
Plan een nieuwe afspraak.
SQL:
INSERT INTO appointment (animal_id, date_time, reason, duration)
VALUES (5, '2026-03-15 10:00', 'Jaarlijkse controle', 30);

24. Situatie:
Koppel een medewerker aan een afspraak.
SQL:
INSERT INTO employee_appointment (employee_id, appointment_id)
VALUES (2, 7);

25. Situatie:
Voeg een technische expertise toe aan een medewerker.
SQL:
INSERT INTO vet_technical_expertise (employee_id, technical_expertise_id)
VALUES (4, 3);

26. Situatie:
Pas de voorraad aan na verkoop.
SQL:
UPDATE product
SET stock_amount = stock_amount - 2
WHERE id = 6;

27. Situatie:
Wijzig het telefoonnummer van een klant.
SQL:
UPDATE customer
SET phone = '0478123456'
WHERE id = 4;

28. Situatie:
Verwijder een afspraak zonder consult.
SQL:
DELETE FROM appointment
WHERE id NOT IN (SELECT id FROM consult);

29. Situatie:
Verwijder een product dat nooit verkocht is.
SQL:
DELETE FROM product
WHERE id NOT IN (
    SELECT product_id FROM purchase
    UNION
    SELECT product_id FROM purchase_at_reception
);

30. Situatie:
Verhoog de prijs van alle producten met 5%.
SQL:
UPDATE product
SET current_price = current_price * 1.05;
