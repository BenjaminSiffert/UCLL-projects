Situatie 1

Geef per klant:

het aantal dieren

het aantal afspraken

het aantal consulten

SELECT customer.first_name, customer.last_name,
       COUNT(DISTINCT animal.id) AS aantal_dieren,
       COUNT(DISTINCT appointment.id) AS aantal_afspraken,
       COUNT(DISTINCT consult.id) AS aantal_consulten
FROM customer
LEFT JOIN animal ON customer.id = animal.customer_id
LEFT JOIN appointment ON animal.id = appointment.animal_id
LEFT JOIN consult ON appointment.id = consult.id
GROUP BY customer.first_name, customer.last_name;

Situatie 2
Geef dieren die wel afspraken hadden maar nog nooit een consult kregen.

SELECT animal.name
FROM animal
JOIN appointment ON animal.id = appointment.animal_id
LEFT JOIN consult ON appointment.id = consult.id
GROUP BY animal.name
HAVING COUNT(appointment.id) > 0
AND COUNT(consult.id) = 0;

Situatie 3

Geef medewerkers die meer afspraken hebben dan het gemiddelde aantal afspraken per medewerker.
SELECT employee.first_name, employee.last_name
FROM employee
JOIN employee_appointment ON employee.id = employee_appointment.employee_id
GROUP BY employee.first_name, employee.last_name
HAVING COUNT(employee_appointment.appointment_id) >
(
    SELECT AVG(aantal)
    FROM (
        SELECT COUNT(employee_appointment.appointment_id) AS aantal
        FROM employee
        LEFT JOIN employee_appointment
            ON employee.id = employee_appointment.employee_id
        GROUP BY employee.id
    ) subquery
);

Situatie 10

Geef medewerkers die meer verschillende klanten bedienen dan het gemiddelde

SELECT employee.first_name, employee.last_name
FROM employee
JOIN employee_appointment ON employee.id = employee_appointment.employee_id
JOIN appointment ON employee_appointment.appointment_id = appointment.id
JOIN animal ON appointment.animal_id = animal.id
GROUP BY employee.first_name, employee.last_name
HAVING COUNT(DISTINCT animal.customer_id) >
(
    SELECT AVG(aantal)
    FROM (
        SELECT COUNT(DISTINCT animal.customer_id) AS aantal
        FROM employee
        JOIN employee_appointment ON employee.id = employee_appointment.employee_id
        JOIN appointment ON employee_appointment.appointment_id = appointment.id
        JOIN animal ON appointment.animal_id = animal.id
        GROUP BY employee.id
    ) subquery
);

Situatie 11

Verhoog de consultprijs met 10% voor dieren met meer dan 3 consulten.

UPDATE consult
SET price = price * 1.10
WHERE id IN (
    SELECT consult.id
    FROM consult
    JOIN appointment ON consult.id = appointment.id
    JOIN animal ON appointment.animal_id = animal.id
    GROUP BY animal.id
    HAVING COUNT(consult.id) > 3
);

Situatie 12

Verwijder afspraken in het verleden zonder medewerkers.

DELETE FROM appointment
WHERE date_time < CURRENT_TIMESTAMP
AND id NOT IN (
    SELECT appointment_id FROM employee_appointment
);

Situatie 13
Verhoog de duur van afspraken met meer dan één medewerker.

UPDATE appointment
SET duration = duration + 15
WHERE id IN (
    SELECT appointment_id
    FROM employee_appointment
    GROUP BY appointment_id
    HAVING COUNT(employee_id) > 1
);
